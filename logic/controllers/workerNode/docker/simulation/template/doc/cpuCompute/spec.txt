//Compile with OpenMP + CPU vector instructions.
gcc mc_openmp.c -O3 -march=native -fopenmp -ffast-math -o mc_sim

//Sample Input
./mc_sim 100000000 100 100 0.05 0.2 1

//Flags and their purpose
O3 => aggressive optimization
march=native => enable AVX/AVX2/AVX512
fopenmp => multithreading
ffast-math => faster floating point
